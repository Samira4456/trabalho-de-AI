#Calculadora
#Funçao para soma
def adicionar(x, y):
    return x + y

#Funçao para subtraçao
def subtrair(x, y):
    return x - y

#Funçao para multiplicaçao
def multiplicar(x, y):
    return x * y

#Funçao para dividir
def dividir(x, y):
    if y != 0:
        return x / y
    else:
        return "Erro! Divisão por zero."
